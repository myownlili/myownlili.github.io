              this.setData({
                searesult:res.data.msg
              })
            }
          })
    },

})

// pages/10/10.js
Page({

    /**
     * 页面的初始数据
     */
    data: {
        content:[]
    },
    createclick:function(){
        wx.request({
            url: 'http://api.tianapi.com/txapi/tiangou/index', //仅为示例，并非真实的接口地址
            data: {
              key: '09d5f0490a9f3044317b3b9a03fd16f0',
            },
            success: (res)=> {
                this.setData({
                 content: res.data.newslist[0].content
                })
            }
          }),
          wx.showToast({
            title: '成功',
            icon: 'success',
            duration: 2000
          })
    },


    /**
     * 生命周期函数--监听页面加载
     */
    onLoad: function (options) {
        wx.showModal({
            title: '爱的提示',
            content: '做自己的舔狗，让别人无处可舔',
            success (res) {
              if (res.confirm) {
                console.log('用户点击确定')
              } else if (res.cancel) {
                console.log('用户点击取消')
              }
            }
          })
    }
})
